Checkboxes and radio addons

Place any checkbox or radio option within an input group's addon instead of text.

<div class="grid two">
  <div class="column">
    <div class="inputs">
      <span class="addon">
        <input type="checkbox">
      </span>
      <input type="text">
    </div>
  </div>

  <div class="column">
    <div class="inputs">
      <span class="addon">
        <input type="radio">
      </span>
      <input type="text">
    </div>
  </div>
</div>
